public class ejecucion {
   
 public static void main(String[] args) {

        int contador = 1;
        int suma = 0;

        while (contador <= 10) {


            suma = suma + contador;
            contador++;


        }
        System.out.println("La sumatoría de los números enteros comprendidos entre el 1 y el 10 es:  " + suma);

    }
    
}
